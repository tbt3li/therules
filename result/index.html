How It Works

  1. The system checks if the requested path is in the list of valid paths
  2. Paths with trailing slashes (like "/result/") are normalized and treated the same as without slashes
  3. If the path is valid (like "/result" or "/result/"), the page loads normally
  4. If the path is invalid (like "/newresult"), the user is redirected to the home page
  5. The redirection happens seamlessly with a visual indicator
// Valid paths array for thefamily.page.dev const validPaths = ['result', 'about', 'contact', 'services', 'family', 'members']; // Function to normalize path (remove trailing slashes) function normalizePath(path) { return path.replace(/\/+$/, ''); // Remove trailing slashes } // Function to check if path is valid function isValidPath(path) { const normalizedPath = normalizePath(path); return validPaths.includes(normalizedPath); } // Function to handle navigation function handleNavigation() { // Get current path (remove leading slash) const path = window.location.pathname.substring(1); // Clear any previous status messages document.getElementById('statusMessage').className = 'status'; document.getElementById('statusMessage').innerHTML = ''; document.getElementById('redirectAnimation').style.display = 'none'; if (!path) { // This is the home page document.getElementById('statusMessage').className = 'status valid'; document.getElementById('statusMessage').innerHTML = 'You are on the home page.'; return; } if (isValidPath(path)) { // Valid path const normalizedPath = normalizePath(path); document.getElementById('statusMessage').className = 'status valid'; document.getElementById('statusMessage').innerHTML = `Valid path: "/${path}" (normalized to "/${normalizedPath}") - Page would load normally.`; } else { // Invalid path document.getElementById('statusMessage').className = 'status invalid'; document.getElementById('statusMessage').innerHTML = `Invalid path: "/${path}" - You will be redirected to the home page.`; // Show redirect animation document.getElementById('redirectAnimation').style.display = 'block'; // Redirect to home page after a short delay setTimeout(() => { window.location.href = 'https://thefamily.page.dev/'; }, 2000); } }
